%general macro of a box
\newcommand{\caja}[7]{
	\color{background}
	\begin{textblock}{#3}(#1, #2)
		% Border color is:
		%	'background' if border = 0px
		%	'border' if border is larger than = 0px
		\pgfsetstrokecolor{#7}
		\pgfsetlinewidth{#5mm}
		\pgfsetcornersarced{\pgfpoint{#6mm}{#6mm}}
		\pgfpathrectangle{\pgfpoint{0mm}{0mm}}{\pgfpoint{#3mm}{#4mm}}
		\pgfusepath{fill,stroke}
	\end{textblock}
}

%general macro of a margin
\newcommand{\margen}[5]{
	\color{margin}
	\begin{textblock}{#3}(#1, #2)
		\pgfsetstrokecolor{margin}
		\pgfpathrectangle{\pgfpoint{0mm}{0mm}}{\pgfpoint{#3mm}{#4mm}}\pgfsetlinewidth{#5mm}
		\pgfusepath{fill,stroke}
	\end{textblock}
}

%general macro of a paragraph
\newcommand{\parrafo}[7]{
	\begin{textblock}{#3}(#1, #2)

			\noindent#4
		\end{#5}
	\end{textblock}
}

%general macro of a rotated paragraph
\newcommand{\parraforrotado}[8]{
	\begin{textblock}{#4}(#1, #2)
		\rotatebox[origin=c]{#3}{
			\begin{#6}
				\fontsize{#7}{#8}\selectfont
				\noindent#5
			\end{#6}
		}
	\end{textblock}
}
